AQS源码学习(ReentrantLock)

  1. ReentrantLock内部抽象类Sync，以及实现公平锁和非公平锁的两个实现类，公平锁实现公平的原理就是多一个队列前面是否有其他线程等待的判断

    abstract static class Sync extends AbstractQueuedSynchronizer {
    abstract void lock();
    //...
}

static final class NonfairSync extends Sync {
    final void lock() {
        //每次需要上锁就尝试修改state值，抢占成功则设置占有锁线程
        if (compareAndSetState(0, 1))
            setExclusiveOwnerThread(Thread.currentThread());
        else
            acquire(1);//AQS方法判断是否需要入队
    }
    //尝试获取锁
    protected final boolean tryAcquire(int acquires) {
        final Thread current = Thread.currentThread();
        int c = getState();
        //state值为0代表资源无人占有，当前线程直接抢占
        if (c == 0) {
            if (compareAndSetState(0, acquires)) {
                setExclusiveOwnerThread(current);
                return true;
            }
        }
        //state不为0则获取线程信息，如果和当前线程是同一线程，调整state值即可
        //即为重入锁次数+1
        else if (current == getExclusiveOwnerThread()) {
            int nextc = c + acquires;
            if (nextc < 0) // overflow
                throw new Error("Maximum lock count exceeded");
            setState(nextc);
            return true;
        }
        //获取失败
        return false;
    }
}
static final class FairSync extends Sync {
    
    final void lock() {
        acquire(1);//AQS方法判断是否需要入队
    }
    //尝试获取锁
    protected final boolean tryAcquire(int acquires) {
        final Thread current = Thread.currentThread();
        int c = getState();
        if (c == 0) {
            //多了一个判断，如果有其他线程已经在等待，当前线程会进入队列等待
            if (!hasQueuedPredecessors() &&
                compareAndSetState(0, acquires)) {
                setExclusiveOwnerThread(current);
                return true;
            }
        }
        else if (current == getExclusiveOwnerThread()) {
            int nextc = c + acquires;
            if (nextc < 0)
                throw new Error("Maximum lock count exceeded");
            setState(nextc);
            return true;
        }
        return false;
    }
}
    1. lock流程总结：发现资源已被占用，抢占失败时，检查队列是否为空，为空则建立虚拟节点并把当前节点连在后面，不为空则把节点置于队尾，同时把前置节点的waitstatus置-1，表示随时可以唤醒；如果出现异常或需要离开队列，则执行删除节点操作
    2. 注意：线程在lock操作后，未获得锁之前，处于阻塞状态，直至唤醒才离开循环
    //AQS内的方法，用于判断是否入队以及队列操作
public final void acquire(int arg) {
    //1.尝试获取锁
    //2.获取失败则入队
    //3.入队失败则中断线程
    if (!tryAcquire(arg) &&
        acquireQueued(addWaiter(Node.EXCLUSIVE), arg))
        selfInterrupt();
}

//尝试把节点放入队列末尾
private Node addWaiter(Node mode) {
    //获取当前队列节点，Exclusive排他模式
    Node node = new Node(Thread.currentThread(), mode);
    //把当前节点放入队列末尾，对应方法比较简单
    Node pred = tail;
    if (pred != null) {
        node.prev = pred;
        if (compareAndSetTail(pred, node)) {
            pred.next = node;
            return node;
        }
    }
    enq(node);
    return node;
}

final boolean acquireQueued(final Node node, int arg) {
    boolean failed = true;
    try {
        boolean interrupted = false;
        for (;;) {
            final Node p = node.predecessor();
            //当前节点为队列第一个节点且抢占成功，重置头节点为当前节点
            if (p == head && tryAcquire(arg)) {
                setHead(node);
                p.next = null; // help GC
                failed = false;
                return interrupted;
            }
            //抢占失败并且当前节点并不是队列第一位，则完成下面两个操作
            //1.修改前置节点的waitstatus值为-1，说明前置节点线程随时可以接收资源
            //2.阻塞当前线程等待唤醒
            if (shouldParkAfterFailedAcquire(p, node) &&
                parkAndCheckInterrupt())
                interrupted = true;
        }
    } finally {
        //出现异常被中断等情况
        if (failed)
            //取消当前队列节点，对应操作就是链表删除节点的操作
            cancelAcquire(node);
    }
}
    1. unlock流程总结：首先清空资源对应锁内的占有线程信息，将state置0表示资源未被占用，然后将队列第一个线程信息取出，用于抢占资源，同时把队列对应节点清空，作为新的虚拟队列头，原虚拟队列头删除；如果前面的节点为空或线程还没准备好，则找到第一个准备好的线程将其唤醒
    public void unlock() {
    sync.release(1);
}

public final boolean release(int arg) {
    if (tryRelease(arg)) {
        Node h = head;
        //唤醒第一个线程
        if (h != null && h.waitStatus != 0)
            unparkSuccessor(h);
        return true;
    }
    return false;
}

protected final boolean tryRelease(int releases) {
    int c = getState() - releases;
    //当前线程不为持有锁的线程，则抛出异常
    if (Thread.currentThread() != getExclusiveOwnerThread())
        throw new IllegalMonitorStateException();
    boolean free = false;
    //清除锁内线程信息，state置0
    if (c == 0) {
        free = true;
        setExclusiveOwnerThread(null);
    }
    setState(c);
    return free;
}

private void unparkSuccessor(Node node) {
    //头节点waitstatus置0
    int ws = node.waitStatus;
    if (ws < 0)
        compareAndSetWaitStatus(node, ws, 0);
    //前面的节点为空或线程还没准备好，则找到第一个准备好的线程
    Node s = node.next;
    if (s == null || s.waitStatus > 0) {
        s = null;
        for (Node t = tail; t != null && t != node; t = t.prev)
            if (t.waitStatus <= 0)
                s = t;
    }
    //唤醒对应线程
    if (s != null)
        LockSupport.unpark(s.thread);
}
