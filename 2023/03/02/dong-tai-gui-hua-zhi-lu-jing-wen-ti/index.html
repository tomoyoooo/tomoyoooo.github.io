动态规划之路径问题

1.动态规划的基础思路

  1. 状态是否具有后效性：也就是对于某个状态来说，我们只需要关注前一步状态到当前状态是如何转化的，而不需要关心之前的状态，就是无后效性的。

  2. 如何定义状态：一般和结尾、答案等是有关联的。(比如最小路径和的dp数组代表的就是当前最小的路径和)

  3. 如何确定状态转移方程：可以从最后一步来进行分类讨论。

  4. 对状态转移的要求

    1)求最值等问题中，只需要要求不漏，因为重复不影响结果

    2)求方案数等问题中，则需要不重不漏

  5. 动态规划的时间复杂度往往和空间复杂度相关

2.记录经历的路径

//注意点1：倒着记录，所以存储好之后要逆序处理一次
//注意点2：二维网格每个点的idx处理可以只利用一维数组
class Solution {
    int m, n;
    public int minPathSum(int[][] grid) {        
        m = grid.length;
        n = grid[0].length;
        int[][] f = new int[m][n];
        int[] g = new int[m * n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (i == 0 && j == 0) {
                    f[i][j] = grid[i][j];
                } else {
                    int top  = i - 1 >= 0 ? f[i - 1][j] + grid[i][j] : Integer.MAX_VALUE;
                    int left = j - 1 >= 0 ? f[i][j - 1] + grid[i][j] : Integer.MAX_VALUE;
                    f[i][j] = Math.min(top, left);
                    g[getIdx(i, j)] = top < left ? getIdx(i - 1, j) : getIdx(i, j - 1);
                }
            }
        }
        
        // 从「结尾」开始，在 g[] 数组中找「上一步」
        int idx = getIdx(m - 1, n - 1);
        // 逆序将路径点添加到 path 数组中
        int[][] path = new int[m + n][2];
        path[m + n - 1] = new int[]{m - 1, n - 1};
        for (int i = 1; i < m + n; i++) {
            path[m + n - 1 - i] = parseIdx(g[idx]);
            idx = g[idx];
        }
        // 顺序输出位置
        for (int i = 1; i < m + n; i++) {
            int x = path[i][0], y = path[i][1];
            System.out.print("(" + x + "," + y + ") ");
        }
        System.out.println(" ");
        
        return f[m - 1][n - 1];
    }
    int[] parseIdx(int idx) {
        return new int[]{idx / n, idx % n};
    }
    int getIdx(int x, int y) {
        return x * n + y;
    }
}

        //考虑到只会用到最小值和次小值
        int n = grid.size();
        if(n==1) return grid[0][0];
        vector<vector<int>> dp(n, vector<int>(n));
        int pre = -1;
        int minn = -1;

        for(int i = 0; i < n; ++i){
            dp[0][i] = grid[0][i];
            if(dp[0][i]<=(minn==-1 ? INT_MAX : dp[0][minn])){
                pre = minn;
                minn = i;
            }
            else if(dp[0][i]<=(pre==-1 ? INT_MAX : dp[0][pre])){
                pre = i;
            }
        } 

        for(int i = 1; i < n; ++i){
            int mintemp = -1;
            int pretemp = -1;
            for(int j = 0; j < n; ++j){
                //dp[i][j] = INT_MAX;
                if(j==minn){
                    dp[i][j] = dp[i-1][pre] + grid[i][j];
                }
                else{
                    dp[i][j] = dp[i-1][minn] + grid[i][j];
                }
                if(dp[i][j]<=(mintemp==-1 ? INT_MAX : dp[i][mintemp])){
                    pretemp = mintemp;
                    mintemp = j;
                }
                else if(dp[i][j]<=(pretemp==-1 ? INT_MAX : dp[i][pretemp])){
                    pretemp = j;
                }
            }
            minn = mintemp;
            pre = pretemp;
        }
        int ans = INT_MAX;
        for(int i = 0; i < n; ++i){
            ans = min(ans, dp[n-1][i]);
        }
        return ans;
    }
};

5.记忆化搜索

  1. 所谓的记忆化搜索重点就在于对中间结果的记录，使得后续可以直接利用

  2. 注意点

  • 递归函数的出参(返回值)和入参(每轮的形参)
  • 递归函数的出口(有效情况即完成事件，无效情况即终止事件)
  • 最小单元处理逻辑(如何进入到下一轮)
//Leetcode 1575.统计所有可行路径
class Solution {
public:
    vector<vector<int>> cnt;
    const int mod = 1000000007;

    int dfs(vector<int>& locations, int now, int finish, int fuelnow){
        //已经处理过的不再重复处理
        if(cnt[now][fuelnow]!=-1){
            return cnt[now][fuelnow];
        }
        int n = locations.size();
        //油耗尽且未到达终点
        if(fuelnow==0 && now!=finish){
            cnt[now][fuelnow] = 0;
            return 0;
        }
        //油未耗尽但无法继续前进
        bool hasNext = false;
        for(int i = 0; i < n; ++i){
            if(i!=now){
                int cost = abs(locations[i]-locations[now]);
                if(fuelnow >= cost){
                    hasNext = true;
                    break;
                }
            }
        }
        if(fuelnow && !hasNext){
            int temp = cnt[now][fuelnow] = now==finish?1:0;
            return temp;
        }

        //如何进入下一轮，并且进行累加
        int sum = now==finish?1:0;
        for(int i = 0; i < n; ++i){
            if(i!=now){
                int cost = abs(locations[i]-locations[now]);
                if(fuelnow >= cost){
                    sum += dfs(locations, i, finish, fuelnow-cost);
                    sum %= mod;
                }
            }
        }
        cnt[now][fuelnow] = sum;
        //返回路径数量和
        return sum;
    }

    int countRoutes(vector<int>& locations, int start, int finish, int fuel) {
        //dfs
        int n = locations.size();
        cnt.resize(n, vector<int>(fuel+1, -1));

        return dfs(locations, start, finish, fuel);
    }
};
//Leetcode 403.青蛙过河
class Solution {
public:
    unordered_map<int,bool> book;
    unordered_map<int,int> mp;
    vector<int> dir{-1,0,1};
    bool dfs(vector<int>& stones, int k, int index){
        int n = stones.size();
        //相当于将原本需要两个int维度表示的东西合并成一个维度，
        //因此二维数组book可以直接使用哈希表代替
        //原本-1代表false,1代表true，0代表未计算
        //现在不存在即未计算
        int key = index*10000+k;
        if(book.count(key)) return book[key];
        if(index==n-1){
            return true;
        }
        for(int i = 0; i < 3; ++i){
            int newk = k + dir[i];
            if(newk<=0) continue;
            if(mp.count(stones[index]+newk)){
                bool flag = dfs(stones, newk, mp[stones[index]+newk]);
                book[key] = flag;
                if(flag) return true;
            }
        }
        book[key] = false;
        return false;
    }

    bool canCross(vector<int>& stones) {
        int n = stones.size();
        if(stones[1]!=1) return false;
        for(int i = 0; i < n; ++i){
            mp[stones[i]] = i;
        }
        return dfs(stones,1,1);;
    }
};

        return dfs(nums, target, 0, 0);
    }
    */
    
    //转动态规划
    int findTargetSumWays(vector<int>& nums, int target) {
        //dp[i][j] = dp[i-1][j+nums[i-1]] + dp[i-1][j-nums[i-1]];
        int n = nums.size();
        int sum = 0;
        for(int i = 0; i < n; ++i){
            sum += abs(nums[i]);
        }
        if(sum < abs(target)) return 0;
        vector<vector<int>> dp(n+1, vector<int>(2*sum+1));

        dp[0][sum] = 1;
        for(int i = 1; i <= n; ++i){
            int x= nums[i-1];
            for(int j = -sum; j <= sum; ++j){
                if(j-x+sum>=0) dp[i][j+sum] += dp[i-1][j-x+sum];
                if(j+x+sum<=2*sum) dp[i][j+sum] += dp[i-1][j+x+sum];
            }
        }
        return dp[n][target+sum];
    }
};

6.总结

  1. 熟悉的题使用以往经验直接猜测状态定义，然后梳理出状态转移方程
  2. 所有dp题都可以先考虑记忆化搜索的方法，然后根据记忆化搜索的函数形参来确定dp数组的状态定义
