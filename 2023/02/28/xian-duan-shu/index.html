线段树

  1. 线段树有两种操作：更新(时间复杂度O(logn))，区间查询(时间复杂度O(logn));
  2. n个节点，树的最大深度为log2(n-1)往下取整后+2
  3. 使用线段树统计的东西必须符合区间加法，比如数字和，最大公因数，最大值等，不能统计需要连续性的东西
  4. 线段树所需空间：将n向上扩充到最近的2的某个次方，再*2(简单一点就直接4*n的大小)
  5. 线段树是一个完全二叉树，比如给定一个数组[1,5,3,7,3,2,5,7],对应的最小值线段树如下图：



定义与建树

#define maxn 10007;
int Sum[maxn<<2]; //4倍空间
int Add[max<<2]; //懒惰标记
int A[maxn],n;

// 2*v -> v<<1 ; 2*v+1 -> v<<1|1
// 求和函数
void PushUp(int rt)
{
  Sum[rt] = Sum[rt<<1] + Sum[rt<<1|1];
}

//[l,r]表示当前节点区间，rt表示当前节点的实际存储位置
//本示例为求和
void Build(int l, int r, int rt)
{
  if(l==r){
    Sum[rt] = A[l];
    return;
  }
  int m = l + (r-l)>>1;
  Build(l,m,rt<<1);
  Build(m+1,r,rt<<1|1);
  PushUp(rt);
}

单点修改:A[L]+=C

//从叶节点修改到本节点
void Update(int L, int C, int l, int r, int rt)
{
  if(l==r){
    Sum[rt] += C;
    return;
  }
  int m = (l+r)>>1;
  if(L<=m) Update(L,C,l,m,rt<<1);
  else Update(L,C,l,m,rt<<1|1);
  PushUp(rt);
}

区间查询:A[L…R]

//[1,n]的线段树可以将任意子区间[L,R]分解为不超过2*log2(n-1)向下取整个子区间
//[L,R]为操作区间，[l,r]为当前区间
int Query(int L, int R, int l, int r, int rt)
{
  if(L<=l && r<=R){
    return Sum[rt];
  }
  int m = (l+r)>>1;
  PushDown(rt, m-l+1, r-m);
  int ans = 0;
  if(L<=m) ans += Query(L,R,l,m,rt<<1);
  if(R>=m) ans += Query(L,R,m+1,r,rt<<1|1);
  return ans;
}

区间修改:A[L,R]+=C

void Update(int L, int R, int C, int l, int r, int rt)
{
  if(L<=l && r<=R){
    Sum[rt] += C*(r-l+1);
    Add[rt] += C;
    return;
  }
  int m = (l+r)>>1;
  PushDown(rt, m-l+1, r-m);
  if(L<=m) Update(L,R,C,l,m,rt<<1);
  if(R>=m) Update(L,R,C,m+1,r,rt<<1|1);
  PushUp(rt); 
}

//ln和rn为左子树和右子树的数字数量
void PushDown(int rt, int ln, int rn)
{
  if(Add[rt]){
    Add[rt<<1] += Add[rt];
    Add[rt<<1|1] += Add[rt];
    Sum[rt<<1] += Add[rt]*ln;
    Sum[rt<<1|1] += Add[rt]*rn;
    Add[rt] = 0; 
  }
}

leetcode 307.区域和检索-数组可修改

class NumArray {
public:
    vector<int> segmentTree;
    int n;
    void build(int node, int l, int r, vector<int> &nums){
        if(l==r){
            segmentTree[node]= nums[l];
            return;
        }
        int m = l + (r-l)/2;
        build(node*2+1, l, m, nums);
        build(node*2+2, m+1, r, nums);
        segmentTree[node] = segmentTree[node*2+1] + segmentTree[node*2+2];
    }

    void change(int index, int val, int node, int l, int r){
        if(l==r){
            segmentTree[node] = val;
            return;
        }
        int m = l + (r-l)/2;
        if(index <= m){
            change(index, val, node*2+1, l, m);
        }
        else{
            change(index, val, node*2+2, m+1, r);
        }
        segmentTree[node] = segmentTree[node*2+1] + segmentTree[node*2+2];
    }

    int range(int left, int right, int node, int l, int r){
        if(left==l && right==r){
            return segmentTree[node];
        }
        int m = l + (r-l)/2;
        if(right <= m){
            return range(left, right, node*2+1, l, m);
        }
        else if(left > m){
            return range(left, right, node*2+2, m+1, r);
        }
        else{
            return range(left, m, node*2+1, l, m) + range(m+1, right, node*2+2, m+1, r);
        }
    }
    

    NumArray(vector<int>& nums) {
        n = nums.size();
        segmentTree.resize(n*4);
        build(0,0,n-1,nums);
    }
    
    void update(int index, int val) {
        change(index, val, 0, 0, n-1);
    }
    
    int sumRange(int left, int right) {
        return range(left, right, 0, 0, n-1);
    }
};

leetcode 731.我的日程安排表 II

class MyCalendarTwo {
public:
    MyCalendarTwo() {

    }

    void update(int start, int end, int val, int l, int r, int idx) {
        if(start>r || end<l) return;
        else if(start<=r && end>=l){
            tree[idx].first += val;
            tree[idx].second += val;
        }
        else{
            int mid = (l+r) >> 1;
            update(start, end-1, 1, l, mid, 2*idx);
            update(start, end-1, 1, mid+1, r, 2*idx+1);
            tree[idx].first = tree[idx].second + max(tree[2*idx].first, tree[2*idx+1].first);
        }

    }

    bool book(int start, int end) {            
        update(start, end-1, 1, 1, 1e9, 1);
        if(tree[1].first > 2){
            update(start, end-1, -1, 1, 1e9, 1);
            return false;
        }
        return true;
    }
private:
    unordered_map<int, pair<int, int>> tree;
};
