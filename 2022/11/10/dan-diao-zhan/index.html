单调栈

一、原理

  1. 将栈内小于/大于自己的元素排除，由此栈内元素一定保持单调性。

  2. 复杂度：只需要遍历原序列一次(常数次)即可，故复杂度为O(n)。

二、能解决的问题

  1. 找到离自己最近的比自己大/小的元素：常用来找峰值、得到有序子序列等。

三、例题

leetcode 654.最大二叉树(c++)

class Solution {
public:
    TreeNode* constructMaximumBinaryTree(vector<int>& nums) {
        int n = nums.size();
        stack<int> st;
        vector<int> left(n, -1);
        vector<int> right(n, -1);
        vector<TreeNode*> tree(n);
        for(int i = 0; i < n; ++i){
            tree[i] = new TreeNode(nums[i]);
            while(!st.empty() && nums[st.top()] < nums[i]){
                right[st.top()] = i;
                st.pop();
            }
            if(!st.empty()){
                left[i] = st.top();
            }
            st.push(i);
        }
        //cout << 1 << endl;
        TreeNode* root = nullptr;
        for(int i = 0; i < n; ++i){
            if(left[i]==-1 && right[i]==-1){
                root = tree[i];
            }
            else if(right[i]==-1 || (left[i]!=-1 && right[i]!=-1 && nums[left[i]] < nums[right[i]])){
                tree[left[i]]->right = tree[i];
            }
            else{
                tree[right[i]]->left = tree[i];
            }
        }
        return root;
    }
};

Leetcode 496.下一个更大元素 I

class Solution {
    public int[] nextGreaterElement(int[] nums1, int[] nums2) {
        Map<Integer, Integer> mp = new HashMap<Integer, Integer>();
        //Stack<Integer> st = new Stack<Integer>();
        Deque<Integer> st = new LinkedList<>();
        for(int i = nums2.length-1; i>=0; --i){
            while(!st.isEmpty() && st.peekFirst()<nums2[i]){
                st.removeFirst();
            }
            if(st.isEmpty()){
                mp.put(nums2[i], -1);
            }
            else mp.put(nums2[i], st.peek());
            st.addFirst(nums2[i]);
        }

        int[] ans = new int[nums1.length];
        for(int i = 0; i < nums1.length; ++i){
            ans[i] = mp.get(nums1[i]);
        }
        return ans;
    }
}

Leetcode 739.每日温度

class Solution {
    public int[] dailyTemperatures(int[] temperatures) {
        //Stack<Integer> st = new Stack<>();
        Deque<Integer> st = new LinkedList<>();
        int[] ans = new int[temperatures.length];
        for(int i = temperatures.length-1; i>=0; --i){
            while(!st.isEmpty() && temperatures[st.peekFirst()] <= temperatures[i]){
                st.removeFirst();
            }
            ans[i] = st.isEmpty() ? 0 : st.peekFirst()-i;
            st.addFirst(i);
        }
        return ans;
    }
}

Leetcode 316.去除重复字母

class Solution {
    public String removeDuplicateLetters(String s) {
        int[] lastidx = new int[26];
        int n = s.length();
        for(int i = 0; i < n; ++i){
            int tmp = s.charAt(i)-'a';
            lastidx[tmp] = i;
        }
        boolean[] appeared = new boolean[26];
        //Stack<Character> st = new Stack<>();
        Deque<Character> st = new LinkedList<>();
        for(int i = 0; i < n; ++i){
            char c = s.charAt(i);
            if(appeared[c-'a']) continue;
            // 把前面比自己字母序大的字母(后面还会继续出现)删除，如果此次是该字母最后一次出现，则直接使用
            while(!st.isEmpty() && st.peekLast()>c && lastidx[st.peekLast()-'a']>i){
                appeared[st.peekLast()-'a'] = false;
                st.removeLast();
            }
            st.addLast(c);
            appeared[c-'a'] = true;
        } 
        StringBuilder sb = new StringBuilder();
        for(Character c : st){
            sb.append(c);
        }
        return sb.toString();
    }
}
